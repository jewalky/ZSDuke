class AlienTrooper : DukeMonster
{
	Default
	{
		Monster;
		Radius 24;
		Height 64;
		Health 100;
		Mass 100;
		PainChance 64;
		Speed 6;
		
		Scale 0.7;
		
		Obituary "%o got shot by a Predator Trooper.";
		
		/*SeeSound "alien/see";
		PainSound "alien/pain";
		DeathSound "alien/death";
		ActiveSound "alien/roam";*/
		
		Translation "112:127=[255,255,0]:[0,0,0]";
	}

	States
	{
	Spawn:
		LIZT AAAABBBB 8 A_Look;
		loop;
		
	See:
		LIZT AAAABBBBCCCCDDDD 1 A_Chase;
		loop;
		
	Missile:
		LIZT E 4 { A_FaceTarget(); A_PlaySound("", CHAN_AUTO); }
		LIZT F 4 BRIGHT { A_FaceTarget(); A_SpawnProjectile("ShrinkRay"); }
		LIZT E 2;
		goto See;
	
	Pain:
		LIZT I 8 A_Pain;
		goto See;
		
	Death:
		TNT1 A 0 A_Scream;
		LIZT IJK 7;
		LIZT L 7 A_NoBlocking;
		LIZT M -1 A_SetFloorClip;
		stop;
		
	XDeath:
		TNT1 A 0 A_ThrowGibs;
		stop;
	}
}
